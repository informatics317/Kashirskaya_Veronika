!define table(x) class x << (T,#FFAAAA) >>
!define pr_key(x) <u>x</u>
!define fr_key(x) <i>x</i>

table(Игроки) {
  pr_key(ID игрока)
  Имя игрока
}

table(Персонажи) {
  pr_key(ID персонажа)
  fr_key(ID игрока)
  Имя персонажа
  Класс
  hp
  mp
  strength
  iq
  damage
}

table(Артефакты) {
  pr_key(ID артефакта)
  Название
  Тип
  Эффект
}

table(Монстры) {
  pr_key(ID монстра)
  Имя монстра
  hp
  mp
  strength
  damage
}

  pr_key(ID битвы)
  fr_key(ID монстра)
  Дата
  Результат
  Кол-во раундов
}
